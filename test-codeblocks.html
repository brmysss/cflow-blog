经典的Mac浏览器风格设计，具有三个点装饰和悬停显示复制按钮

JavaScript 代码块
javascript
function addCodeBlockCopyButtons() {
    const codeBlocks = document.querySelectorAll('.memos__text pre');
    
    codeBlocks.forEach((pre, index) => {
        // 检查是否已经有复制按钮
        if (pre.querySelector('.copy-btn')) {
            return;
        }
        
        const code = pre.querySelector('code');
        if (!code) return;
        
        // 创建复制按钮
        const copyBtn = document.createElement('button');
        copyBtn.className = 'copy-btn';
        copyBtn.textContent = '复制';
    });
}
Python 代码块
python
def process_line_breaks(content):
    """智能处理换行符，避免在代码块中插入 """
    code_blocks = []
    protected_content = content.replace(/```[\s\S]*?```/g, function(match):
        placeholder = f"__CODE_BLOCK_{len(code_blocks)}__"
        code_blocks.append(match)
        return placeholder
    
    # 在非代码块内容中处理换行
    protected_content = protected_content.replace(/\n\n/g, "\n\n \n\n")
    
    return protected_content
